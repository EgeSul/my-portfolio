CMS Client — Teknik Analiz

Dinamik CMS içeriğinin Redis MemoryCache katmanından asenkron olarak sorgulanması ve HTTP isteklerinin optimize edilmesi.

[C#] Redis Önbellekli Veri İstek Yönetimi
public async Task<List<PostDto>> GetCachedPostsAsync() {
  string cacheKey = "ozancms_posts_ege";
  if (!_cache.TryGetValue(cacheKey, out List<PostDto> posts)) {
    var response = await _httpClient.GetAsync("https://api.ozancms.com/ege/posts");
    if (response.IsSuccessStatusCode) {
      posts = await response.Content.ReadFromJsonAsync<List<PostDto>>();
      var cacheOptions = new MemoryCacheEntryOptions().SetAbsoluteExpiration(TimeSpan.FromMinutes(10));
      _cache.Set(cacheKey, posts, cacheOptions);
    }
  }
  return posts;
}

[C#] JWT Bearer Token Doğrulama Middleware
public async Task InvokeAsync(HttpContext context) {
  if (!context.Request.Headers.TryGetValue("Authorization", out var authHeader)) {
    context.Response.StatusCode = 401;
    await context.Response.WriteAsync("Erişim Reddedildi: Token Eksik.");
    return;
  }
  var token = authHeader.ToString().Split(" ").Last();
  if (!ValidateJwtToken(token)) {
    context.Response.StatusCode = 403;
    await context.Response.WriteAsync("Geçersiz Yetki Sınırı.");
    return;
  }
  await _next(context);
}
